train, test = train_test_split(

    random_state=42,

)

def balance_treatment_groups(data: pd.DataFrame) -> pd.DataFrame:
    """
     treatment 
     uplift 
    """
    count_0 = len(data[data['treatment'] == 0])
    count_1 = len(data[data['treatment'] == 1])
    
    if count_0 > count_1:
        # 
        data_0 = data[data['treatment'] == 0].sample(n=count_1, random_state=42)
        data_1 = data[data['treatment'] == 1]
    else:
        # 
        data_0 = data[data['treatment'] == 0]
        data_1 = data[data['treatment'] == 1].sample(n=count_0, random_state=42)
    
    balanced_data = pd.concat([data_0, data_1], ignore_index=True)
    return balanced_data

print(" treatment ")
print(train['treatment'].value_counts())
train_balanced = balance_treatment_groups(train)
print("")
print(train_balanced['treatment'].value_counts())


# ==============================
# 4.  Two-ModelS-LearnerUplift 
# ==============================
print("\n4 S-Learner  treatment ...")

#  f0 ~ f11 
feature_cols = [f'f{i}' for i in range(12)]

X_train = train_balanced[feature_cols].copy()
X_test = test[feature_cols].copy()

y_train = train_balanced['visit']  #  conversion
t_train = train_balanced['treatment']
t_test = test['treatment']

#  treatment S-Learner 
X_train_with_t = X_train.copy()
X_train_with_t['treatment'] = t_train

#  XGBoost 
model = xgb.XGBRegressor(
    n_estimators=200,
    max_depth=6,
    learning_rate=0.1,
    random_state=42,
    n_jobs=-1
)
model.fit(X_train_with_t, y_train)

# 
X_test_t0 = X_test.copy()
X_test_t0['treatment'] = 0  # 

X_test_t1 = X_test.copy()
X_test_t1['treatment'] = 1  # 

pred_control = model.predict(X_test_t0)   # P(visit | not treated)
pred_treated = model.predict(X_test_t1)   # P(visit | treated)

#  uplift score
uplift_scores = pred_treated - pred_control


# ==============================
# 5. Qini  Top-K 
# ==============================
print("\n5...")

#  DataFrame
eval_df = pd.DataFrame({
    'uplift': uplift_scores,
    'treatment': t_test.values,
    'visit': test['visit'].values
}).sort_values('uplift', ascending=False).reset_index(drop=True)

#  visit 
is_visit_in_treated = (eval_df['treatment'] == 1) & (eval_df['visit'] == 1)
is_visit_in_control = (eval_df['treatment'] == 0) & (eval_df['visit'] == 1)

cum_visit_treated = is_visit_in_treated.cumsum()
cum_visit_control = is_visit_in_control.cumsum()

# 
n_treated = eval_df['treatment'].sum()
n_control = len(eval_df) - n_treated
ratio = n_treated / n_control if n_control > 0 else 1

# Incremental Visits
incremental_visits = cum_visit_treated - cum_visit_control * ratio

# 
x_prop = np.arange(1, len(incremental_visits) + 1) / len(incremental_visits)
y_gain = incremental_visits
total_gain = y_gain.iloc[-1]

# 
random_baseline = x_prop * total_gain

#  Qini AUC_model - AUC_random
auc_model = auc(x_prop, y_gain)
auc_random = auc(x_prop, random_baseline)
qini_coeff = auc_model - auc_random

print(f"Qini : {qini_coeff:.4f}")

# Top-K  uplift 
def top_k_capture(k_percent: int) -> float:
    """ Top K% """
    idx = int(len(eval_df) * k_percent / 100)
    captured = incremental_visits.iloc[idx] if idx < len(incremental_visits) else total_gain
    return captured / total_gain if total_gain != 0 else 0

print(f"Top 10% : {top_k_capture(10):.2%}")
print(f"Top 20% : {top_k_capture(20):.2%}")
print(f"Top 30% : {top_k_capture(30):.2%}")
